<script>
let dailyCalories = 0;
let lastCalories = 0;

// load saved goal or default to 2000
let calorieGoal = Number(localStorage.getItem("calorieGoal")) || 2000;
document.getElementById("calorieGoal").textContent = calorieGoal;

/* CHANGE CALORIE GOAL */
function changeGoal() {
    let newGoal = prompt("Enter your new daily calorie goal:");

    if (!newGoal || isNaN(newGoal)) {
        alert("Please enter a valid number.");
        return;
    }

    newGoal = Number(newGoal);

    // save permanently
    localStorage.setItem("calorieGoal", newGoal);

    // update on screen
    document.getElementById("calorieGoal").textContent = newGoal;

    alert("Goal updated!");
}

/* REAL DIARY SAVE */
function saveToDiary(foodName, kcal) {
    let diary = JSON.parse(localStorage.getItem("diary")) || [];

    diary.push({
        name: foodName,
        calories: kcal,
        time: new Date().toLocaleString()
    });

    localStorage.setItem("diary", JSON.stringify(diary));
}


function flashRed() {
    document.body.classList.remove("flash-red");
    void document.body.offsetWidth; // restart animation
    document.body.classList.add("flash-red");
}
function flashGreen() {
    document.body.classList.remove("flash-green");
    void document.body.offsetWidth;
    document.body.classList.add("flash-green");
}

let codeReader = new ZXing.BrowserMultiFormatReader();


/* LIVE CAMERA (online only) */
async function startLiveScan() {
    const preview = document.getElementById("cameraPreview");

    try {
        preview.style.display = "block";

        const devices = await ZXing.BrowserMultiFormatReader.listVideoInputDevices();
        const backCamera = devices[devices.length - 1].deviceId;

        codeReader.decodeFromVideoDevice(backCamera, "cameraPreview", (result, err) => {
            if (result) {
                preview.style.borderColor = "#00ff66";
                preview.style.boxShadow = "0 0 20px #00ff66";
                lookupBarcode(result.text);
                codeReader.reset();
            } else {
                preview.style.borderColor = "#ff4444";
                preview.style.boxShadow = "0 0 20px #ff4444";
            }
        });

    } catch (err) {
        alert("Live camera only works when the app is online (HTTPS).");
    }
}

/* TAKE PHOTO (offline) */
function takePhoto() {
    const input = document.createElement("input");
    input.type = "file";
    input.accept = "image/*";
    input.capture = "environment";

    input.onchange = async (event) => {
        const file = event.target.files[0];
        if (!file) return;

        const reader = new FileReader();
        reader.onload = () => decodeImage(reader.result);
        reader.readAsDataURL(file);
    };

    input.click();
}

/* UPLOAD PHOTO */
function handleUpload(event) {
    const file = event.target.files[0];
    if (!file) return;

    const reader = new FileReader();
    reader.onload = () => decodeImage(reader.result);
    reader.readAsDataURL(file);
}

/* ZXING DECODE */
function decodeImage(dataUrl) {
    codeReader.decodeFromImageUrl(dataUrl)
        .then(result => lookupBarcode(result.text))
        .catch(() => alert("No barcode detected."));
}

/* LOOKUP FOOD */
async function lookupBarcode(code) {
    const url = `https://world.openfoodfacts.org/api/v0/product/${code}.json`;

    const response = await fetch(url);
    const data = await response.json();

    if (!data.product) {
        alert("Food not found.");
        return;
    }

    const p = data.product;
    const n = p.nutriments || {};

    const kcal    = n["energy-kcal_100g"] || 0;
    const fat     = n.fat_100g || 0;
    const sugar   = n.sugars_100g || 0;
    const salt    = n.salt_100g || 0;
    const protein = n.proteins_100g || 0;
    const carbs   = n.carbohydrates_100g || 0;

    lastCalories = Number(kcal);

    /* TRAFFIC LIGHT FUNCTION + LABELS */
    function trafficLight(value, low, high) {
        if (value > high) return { cls: "red", label: "HIGH 🟥" };
        if (value > low)  return { cls: "orange", label: "MED 🟧" };
        return { cls: "green", label: "LOW 🟩" };
    }

    const fatTL   = trafficLight(fat,   3, 17);
    const sugarTL = trafficLight(sugar, 5, 22);
    const saltTL  = trafficLight(salt,  0.3, 1.5);

    const fatSpan   = document.getElementById("fatLight");
    const sugarSpan = document.getElementById("sugarLight");
    const saltSpan  = document.getElementById("saltLight");

    fatSpan.className   = "traffic " + fatTL.cls;
    sugarSpan.className = "traffic " + sugarTL.cls;
    saltSpan.className  = "traffic " + saltTL.cls;

    fatSpan.textContent   = fatTL.label;
    sugarSpan.textContent = sugarTL.label;
    saltSpan.textContent  = saltTL.label;

    /* Fill card */
    document.getElementById("foodName").textContent = p.product_name || "Unknown";
    document.getElementById("calNumber").textContent = kcal;
    document.getElementById("fat").textContent = fat + " g";
    document.getElementById("sugar").textContent = sugar + " g";
    document.getElementById("salt").textContent = salt + " g";

    document.getElementById("productImage").src =
        p.image_front_small_url || p.image_front_url || "https://via.placeholder.com/120";

    /* Healthy Score (NutriScore-based) */
    let score = 50;
    const grade = p.nutriscore_grade;

    if (grade === "a") score = 90;
    else if (grade === "b") score = 75;
    else if (grade === "c") score = 60;
    else if (grade === "d") score = 40;
    else if (grade === "e") score = 20;

    const gauge = document.getElementById("healthGauge");
    gauge.textContent = score;

    let color = "#4caf50";
    if (score < 50) color = "#ef5350";
    else if (score < 70) color = "#ffca28";

    gauge.style.borderColor = color;
    gauge.style.boxShadow = "0 0 12px " + color + "aa";

    /* Macro bars */
    const total = protein + carbs + fat || 1;
    document.getElementById("macroProtein").style.width = (protein / total * 100) + "%";
    document.getElementById("macroCarbs").style.width   = (carbs   / total * 100) + "%";
    document.getElementById("macroFat").style.width     = (fat     / total * 100) + "%";

    /* Rating + Reason (stricter) */
    let rating = "ok";
    let ratingText = "OK choice";
    let reason = "";

    if (fat > 30 || sugar > 15 || salt > 1 || kcal > 400) {
flashRed();

        rating = "bad";
        ratingText = "Not a healthy choice.";

        if (fat > 30) reason += "Extremely high in fat. ";
        if ((n["saturated-fat_100g"] || 0) > 5) reason += "High in saturated fat. ";
        if (sugar > 15) reason += "High in sugar. ";
        if (salt > 1) reason += "High in salt. ";
        if (kcal > 400) reason += "Very high in calories. ";
    } else if (grade === "a" || grade === "b") {
flashGreen();

        rating = "good";
        ratingText = "Good choice!";
        reason = "Low in unhealthy nutrients.";
    } else {
        rating = "ok";
        ratingText = "Moderate choice.";
        if (!reason) reason = "Acceptable in moderation. Watch portion size.";
    }

    document.getElementById("ratingText").textContent = ratingText;
    document.getElementById("ratingText").className = "rating " + rating;
    document.getElementById("reasonText").textContent = reason;

    /* NOVA ultra-processed */
    const nova = p.nova_group || 0;
    const novaEl = document.getElementById("novaWarning");

    if (nova === 4) {
        novaEl.textContent = "Ultra‑processed food (NOVA 4).";
    } else if (nova === 3) {
        novaEl.textContent = "Processed food (NOVA 3).";
    } else if (nova === 2) {
        novaEl.textContent = "Processed culinary ingredient (NOVA 2).";
    } else if (nova === 1) {
        novaEl.textContent = "Unprocessed or minimally processed (NOVA 1).";
    } else {
        novaEl.textContent = "NOVA group: not available.";
    }

    /* Additives */
    let additives = p.additives_tags || [];
    let eList = additives
        .filter(a => a.startsWith("en:"))
        .map(a => a.replace("en:", "").toUpperCase());

    if (eList.length > 0) {
        document.getElementById("additivesWarning").textContent =
            "Contains additives: " + eList.join(", ");
    } else {
        document.getElementById("additivesWarning").textContent = "No additives listed or not available.";
    }

    /* Better alternatives */
    let alt = "";
    if (nova === 4) alt += "Try less processed alternatives. ";
    if (fat > 30)   alt += "Choose lighter spreads or low‑fat sauces. ";
    if (sugar > 15) alt += "Try low‑sugar versions. ";
    if (salt > 1)   alt += "Look for low‑salt options. ";

    if (!alt) {
        alt = "This food is acceptable in moderation. Prefer fresh, minimally processed options when possible.";
    }

    document.getElementById("betterAlt").textContent = alt;

    document.getElementById("resultCard").style.display = "block";
}

/* DAILY CALORIE COUNTER + SAVE TO DIARY */
document.getElementById("addDiaryBtn").addEventListener("click", () => {
    dailyCalories += lastCalories;
    document.getElementById("dailyCalories").textContent = Math.round(dailyCalories);

    const foodName = document.getElementById("foodName").textContent;

    saveToDiary(foodName, lastCalories);

    alert("Added to diary!");
});
